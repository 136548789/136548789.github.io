9: 数组插入数组

//es6
//原数组
var a = [1,2];
//要插入的数组
var b = [3,4];
//要插入的位置
var i = 1;
//插入
a.splice(i,0,...b);
//结果 a = [1, 3, 4, 2]
//es5
//原数组
var a = [1,2];
//要插入的数组
var b = [3,4];
//要插入的位置
var i = 1;
//插入
a.splice.apply(a,[i,0].concat(b));
//结果 a = [1, 3, 4, 2]

10:多维数组变一维数组（超过二维）

//第一反应是递归，后面网上找了几个方法
var newArr = []
function toOneArray(arr) {
    for(var i = 0; i< arr.length;i++){
        if(Array.isArray(arr[i])){
            toOneArray(arr[i])
        }else{
            newArr.push(arr[i])
        }
    }
}

微信公众号支付

发表于 2021-10-08 | 分类于 wechat

微信公众号支付相关代码

     function onBridgeReady() {
    WeixinJSBridge.invoke(
      'getBrandWCPayRequest', {
      "appId": dataCost.appId,//公众号名称，由商户传入     
      "timeStamp": dataCost.timeStamp,//时间戳    
      "nonceStr": dataCost.nonceStr, //随机串     
      "package": dataCost.package,
      "signType": dataCost.signType,//微信签名方式
      "paySign": dataCost.paySign //微信签名 
    },
      function (res) {
        if (res.err_msg == "get_brand_wcpay_request:ok") {
          callback("ok")
          //微信团队郑重提示：res.err_msg将在用户支付成功后返回ok，但并不保证它绝对可靠。
        } else if (res.err_msg == "get_brand_wcpay_request:cancel") {
          //取消 
          callback("cancel");
        } else {
          //失败 
          callback("fail", res.err_desc);
        }
      },
      (err) => {
      });
  }
  if (typeof WeixinJSBridge == "undefined") {
    if (document.addEventListener) {
      document.addEventListener('WeixinJSBridgeReady', onBridgeReady, false);
    } else if (document.attachEvent) {
      document.attachEvent('WeixinJSBridgeReady', onBridgeReady);
      document.attachEvent('onWeixinJSBridgeReady', onBridgeReady);
    }
  } else {
    onBridgeReady();
  }
