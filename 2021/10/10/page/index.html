9: 数组插入数组

//es6
//原数组
var a = [1,2];
//要插入的数组
var b = [3,4];
//要插入的位置
var i = 1;
//插入
a.splice(i,0,...b);
//结果 a = [1, 3, 4, 2]
//es5
//原数组
var a = [1,2];
//要插入的数组
var b = [3,4];
//要插入的位置
var i = 1;
//插入
a.splice.apply(a,[i,0].concat(b));
//结果 a = [1, 3, 4, 2]

10:多维数组变一维数组（超过二维）

//第一反应是递归，后面网上找了几个方法
var newArr = []
function toOneArray(arr) {
    for(var i = 0; i< arr.length;i++){
        if(Array.isArray(arr[i])){
            toOneArray(arr[i])
        }else{
            newArr.push(arr[i])
        }
    }
}
